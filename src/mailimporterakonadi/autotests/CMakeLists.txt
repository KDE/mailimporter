macro(add_mailimporterakonadi_unittest _source _additionalSource)

        TEST_NAME ${_name}-akonadi
        NAME_PREFIX "mailimporterakonadi-"
	LINK_LIBRARIES KF5::MailImporterAkonadi Qt::Test Qt::Core Qt::Gui Qt::Widgets
    )
endmacro ()
